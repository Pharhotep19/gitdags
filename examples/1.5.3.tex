\usepackage{gitdags}

\begin{figure}
  \begin{subfigure}[b]{\textwidth}
    \centering
    \begin{tikzpicture}
      % Commit DAG
      \gitDAG[grow right sep = 2em]{
        A -- B -- C 
      };
      \gitbranch
        {master}     % node name and text 
        {above=of C} % node placement
        {C}          % target
      \gitHEAD
        {above=of master} % node placement
        {master}          % target
    \end{tikzpicture}
  \end{subfigure}
\end{figure}
