%     B --> C(C: master branch ends here)
%     C -- experimental branch --> E
%     E --> H(H <- HEAD)
\documentclass[preview]{standalone}
\usepackage{subcaption}
\usepackage{gitdags}
\begin{document}
\begin{figure}
  \begin{subfigure}[b]{\textwidth}
    \centering
    \begin{tikzpicture}
      % Commit DAG
      \gitDAG[grow right sep = 2em]{
        A -- B -- C -- E -- H
      };
      \gitbranch
        {master}     % node name and text 
        {above=of C} % node placement
        {C}          % target
      \gitbranch
        {experimental}     % node name and text 
        {above=of H} % node placement
        {H}          % target
      \gitHEAD
        {above=of experimental} % node placement
        {experimental}          % target
    \end{tikzpicture}
  \end{subfigure}
\end{figure}
\end{document}
