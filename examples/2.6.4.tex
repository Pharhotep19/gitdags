\documentclass[preview]{standalone}

\usepackage{subcaption}
\usepackage{gitdags}

\begin{document}

\begin{figure}
  \begin{subfigure}[b]{\textwidth}
    \centering
    \begin{tikzpicture}
      % Commit DAG
      \gitDAG[grow right sep = 2em]{
        A50 -- A51 -- A75 -- A76 -- A100
      };
      \gitbranch
        {skip 24 commits}     % node name and text
        {above=of A51} % node placement
        {A51}          % target
      \gitbranch
        {skipped 24 commits}     % node name and text
        {above=of A100} % node placement
        {A100}          % target
      \gitbranch
        {GOOD}     % node name and text
        {below=of A50} % node placement
        {A50}          % target
      \gitbranch
        {?}     % node name and text
        {below=of A75} % node placement
        {A75}          % target
      \gitbranch
        {BAD}     % node name and text
        {below=of A100} % node placement
        {A100}          % target
    \end{tikzpicture}
  \end{subfigure}
\end{figure}
